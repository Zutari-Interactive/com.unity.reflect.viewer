using System.IO;
using Newtonsoft.Json;
using SerializableSO;
using UnityEngine;

namespace $NAMESPACE
{
    // [CreateAssetMenu]
    public class $SCRIPTNAME : SerializableSOBase
    {
        #region VARIABLES

        #endregion

        #region PROPERTIES

        #endregion

        #region SERIALIZATION METHODS
        
        public override string GetFilePath()
        {
            return base.GetFilePath();
        }

        public override void WriteToFile()
        {
            File.WriteAllText(GetFilePath(), JsonConvert.SerializeObject(this, Formatting.Indented));
        }

        public override void OverwriteFromFile()
        {
            if (!FileExist()) return;
            JsonConvert.PopulateObject(File.ReadAllText(GetFilePath()), this);
        }

        public override void DeleteFile()
        {
            base.DeleteFile();
        }
        
        public override bool FileExist()
        {
            return base.FileExist();
        }

        #endregion
    }
}
